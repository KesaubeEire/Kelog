    • 倒放

      • 倒放的改进
        因为上一个效果重复创建Tweener对象,使用次数增多会影响性能
        所以这里我们在start里创建对象,有且仅有一次
        然后在开始之时 使用 tweener.Pause();
        在需要时使用 transform.DoPlayForward();
        在需要倒放时使用 transform.DoPlayBackward();
        下面的注释更加详细
      void Start()
     {
       //在start出
         OriPos = panelTransform.localPosition;
         Tweener tweener = panelTransform.DOLocalMove(targetPos(V3), transferTime);
         tweener.Pause();
         //设置不自动销毁
         tweener.SetAutoKill(false);
     }
     public void Onclickk()
     {
         if (!isMove)
         {
             //只播放一次,且播放其身上的所有动画
 //            panelTransform.DOPlay();
             //前放,不止一次,且播放其身上的所有动画
             panelTransform.DOPlayForward();
             isMove = !isMove;
         }
         else
         {
 //            panelTransform.DOLocalMove(OriPos, transferTime);
             // 倒放
             panelTransform.DOPlayBackwards();
             isMove = !isMove;
          }
     }
}

        //默认动画完成之后销毁
